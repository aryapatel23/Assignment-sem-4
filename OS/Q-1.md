//Q1. Write a program to simulate the Priority CPU scheduling algorithm. 

#include<stdio.h>
#include<stdlib.h>
#include<string.h>

#define MAX_PROCESSES 10

// Structure to store process information
typedef struct {
    int pid;                    // Process ID
    int arrival_time;           // Arrival time
    int burst_time;             // CPU Burst time
    int priority;               // Priority (lower number = higher priority)
    int completion_time;        // Completion time
    int turnaround_time;        // Turnaround time
    int waiting_time;           // Waiting time
    int remaining_time;         // For preemptive scheduling
} Process;

// Comparator function to sort by priority
int compare_priority(const void *a, const void *b) {
    Process *p1 = (Process *)a;
    Process *p2 = (Process *)b;
    
    // First sort by arrival time, then by priority
    if (p1->arrival_time != p2->arrival_time)
        return p1->arrival_time - p2->arrival_time;
    return p1->priority - p2->priority;
}

// Non-Preemptive Priority Scheduling
void non_preemptive_priority(Process processes[], int n) {
    printf("\n========== NON-PREEMPTIVE PRIORITY SCHEDULING ==========\n");
    
    // Sort by arrival time and priority
    qsort(processes, n, sizeof(Process), compare_priority);
    
    int current_time = 0;
    float total_wt = 0, total_tat = 0;
    
    printf("\nExecution Order:\n");
    printf("-------------------------------------------\n");
    
    for (int i = 0; i < n; i++) {
        // Process starts at max of current time and arrival time
        if (current_time < processes[i].arrival_time)
            current_time = processes[i].arrival_time;
        
        current_time += processes[i].burst_time;
        processes[i].completion_time = current_time;
        processes[i].turnaround_time = processes[i].completion_time - processes[i].arrival_time;
        processes[i].waiting_time = processes[i].turnaround_time - processes[i].burst_time;
        
        total_wt += processes[i].waiting_time;
        total_tat += processes[i].turnaround_time;
        
        printf("P%d (Priority: %d) | AT: %d | BT: %d | CT: %d\n",
               processes[i].pid, processes[i].priority, processes[i].arrival_time,
               processes[i].burst_time, processes[i].completion_time);
    }
    
    printf("\n========== RESULTS ==========\n");
    printf("%-8s %-10s %-10s %-10s %-10s %-10s\n",
           "PID", "Arrival", "Burst", "Waiting", "Turn-Around", "Priority");
    printf("-----------------------------------------------------------\n");
    
    for (int i = 0; i < n; i++) {
        printf("P%-7d %-10d %-10d %-10d %-10d %-10d\n",
               processes[i].pid, processes[i].arrival_time, processes[i].burst_time,
               processes[i].waiting_time, processes[i].turnaround_time, processes[i].priority);
    }
    
    printf("-----------------------------------------------------------\n");
    printf("Average Waiting Time: %.2f\n", total_wt / n);
    printf("Average Turnaround Time: %.2f\n", total_tat / n);
}

// Preemptive Priority Scheduling
void preemptive_priority(Process processes[], int n) {
    printf("\n========== PREEMPTIVE PRIORITY SCHEDULING ==========\n");
    
    // Create a copy for calculations
    Process temp[MAX_PROCESSES];
    for (int i = 0; i < n; i++) {
        temp[i] = processes[i];
        temp[i].remaining_time = processes[i].burst_time;
    }
    
    int current_time = 0;
    int completed = 0;
    float total_wt = 0, total_tat = 0;
    
    printf("\nGantt Chart:\n");
    printf("-------------------------------------------\n");
    
    while (completed < n) {
        int highest_priority = 999;
        int idx = -1;
        
        // Find process with highest priority (lowest number) available
        for (int i = 0; i < n; i++) {
            if (temp[i].remaining_time > 0 && temp[i].arrival_time <= current_time) {
                if (temp[i].priority < highest_priority) {
                    highest_priority = temp[i].priority;
                    idx = i;
                }
            }
        }
        
        if (idx == -1) {
            // No process available, move to next arrival time
            int next_arrival = 999;
            for (int i = 0; i < n; i++) {
                if (temp[i].remaining_time > 0 && temp[i].arrival_time > current_time) {
                    next_arrival = (temp[i].arrival_time < next_arrival) ? temp[i].arrival_time : next_arrival;
                }
            }
            current_time = next_arrival;
        } else {
            // Execute process for 1 time unit
            printf("P%d [%d-%d] Priority: %d\n", temp[idx].pid, current_time, current_time + 1, temp[idx].priority);
            current_time++;
            temp[idx].remaining_time--;
            
            // If process completed
            if (temp[idx].remaining_time == 0) {
                temp[idx].completion_time = current_time;
                temp[idx].turnaround_time = temp[idx].completion_time - temp[idx].arrival_time;
                temp[idx].waiting_time = temp[idx].turnaround_time - processes[idx].burst_time;
                
                total_wt += temp[idx].waiting_time;
                total_tat += temp[idx].turnaround_time;
                completed++;
            }
        }
    }
    
    printf("\n========== RESULTS ==========\n");
    printf("%-8s %-10s %-10s %-10s %-10s %-10s\n",
           "PID", "Arrival", "Burst", "Waiting", "Turn-Around", "Priority");
    printf("-----------------------------------------------------------\n");
    
    for (int i = 0; i < n; i++) {
        printf("P%-7d %-10d %-10d %-10d %-10d %-10d\n",
               processes[i].pid, processes[i].arrival_time, processes[i].burst_time,
               temp[i].waiting_time, temp[i].turnaround_time, processes[i].priority);
    }
    
    printf("-----------------------------------------------------------\n");
    printf("Average Waiting Time: %.2f\n", total_wt / n);
    printf("Average Turnaround Time: %.2f\n", total_tat / n);
}

// Function to input process details
void input_processes(Process processes[], int *n) {
    printf("========== PRIORITY CPU SCHEDULING ALGORITHM ==========\n");
    printf("Enter number of processes: ");
    scanf("%d", n);
    
    printf("\nEnter details for each process:\n");
    printf("(Arrival Time, Burst Time, Priority)\n");
    printf("Note: Lower priority number = Higher priority\n\n");
    
    for (int i = 0; i < *n; i++) {
        printf("Process %d:\n", i + 1);
        printf("  Arrival Time: ");
        scanf("%d", &processes[i].arrival_time);
        printf("  Burst Time: ");
        scanf("%d", &processes[i].burst_time);
        printf("  Priority (0=highest): ");
        scanf("%d", &processes[i].priority);
        
        processes[i].pid = i + 1;
        processes[i].remaining_time = processes[i].burst_time;
    }
}

// Function to display menu
void display_menu() {
    printf("\n========== SCHEDULING MENU ==========\n");
    printf("1. Non-Preemptive Priority Scheduling\n");
    printf("2. Preemptive Priority Scheduling\n");
    printf("3. Both (Run both algorithms)\n");
    printf("4. Exit\n");
    printf("Choose option: ");
}

int main() {
    Process processes[MAX_PROCESSES];
    Process backup[MAX_PROCESSES];
    int n = 0;
    int choice;
    
    // Input processes
    input_processes(processes, &n);
    
    // Create backup for second run
    memcpy(backup, processes, sizeof(processes));
    
    while (1) {
        display_menu();
        scanf("%d", &choice);
        
        switch (choice) {
            case 1:
                memcpy(processes, backup, sizeof(backup));
                non_preemptive_priority(processes, n);
                break;
                
            case 2:
                memcpy(processes, backup, sizeof(backup));
                preemptive_priority(processes, n);
                break;
                
            case 3:
                memcpy(processes, backup, sizeof(backup));
                non_preemptive_priority(processes, n);
                
                memcpy(processes, backup, sizeof(backup));
                preemptive_priority(processes, n);
                break;
                
            case 4:
                printf("\nExiting...\n");
                exit(0);
                
            default:
                printf("Invalid choice! Try again.\n");
        }
    }
    
    return 0;
}
